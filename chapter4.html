Chapter 4 / 연산자와 제어문

연산자

age = currentYear - birthYear + 1
  • age, currentYear, birthYear, 1은 연산 대상이 되기 때문에 '피연산자'라고 부름
  • '=, -, +'는 연산자
    •
    종류 연산자 설명 예시
    산술 연산자 + 덧셈 / 문자열 연결 5 + 3 → 8, "a" + "b" → "ab"
    - 뺄셈 5 - 3 → 2
    * 곱셈 5 * 2 → 10
    / 나눗셈 10 / 2 → 5
    % 나머지 10 % 3 → 1
    ** 거듭제곱 2 ** 3 → 8
    대입 연산자 = 값 할당 x = 10
    += 더해서 할당 x += 5 → x = x + 5
    -= 빼서 할당 x -= 2 → x = x - 2
    *= 곱해서 할당 x *= 3 → x = x * 3
    /= 나눠서 할당 x /= 2 → x = x / 2
    %= 나머지 할당 x %= 3 → x = x % 3
    비교 연산자 == 값만 비교 (느슨한 비교) 5 == "5" → true
    === 값과 타입 모두 비교 (엄격한 비교) 5 === "5" → false
    != 값만 다르면 true 5 != "5" → false
    !== 값이나 타입 다르면 true 5 !== "5" → true
    > 크다 7 > 5 → true
    <= 작거나 같다 3 <= 3 → true
    논리 연산자 && AND (모두 true일 때 true) true && false → false
    || OR (하나라도 true면 true) true || false → true
    ! NOT (논리 부정) !true → false

    연산자 우선 순위

    단항 연산자 > 산술 연산자 > 비교 연산자 > 논리 연산자 > 할당 연산자

    제어문

    종류 형식 설명
    if 문 
    if (조건) {
  // 조건이 true일 때 실행
}
    		조건이 참일 때 블록 내부 코드 실행
    if ~ else 
    if (조건) {
  // 참일 때
} else {
  // 거짓일 때
}
    		조건이 거짓이면 else 블록 실행
    if ~ else if 
    if (조건1) {
  // 조건1이 true
} else if (조건2) {
  // 조건2가 true
} else {
  // 모두 false일 때
}
    		여러 조건을 순차적으로 비교
    switch 
    switch (변수) {
  case 값1:
    // 값1일 때 실행
    break;
  case 값2:
    // 값2일 때 실행
    break;
  default:
    // 아무 case도 아닐 때
}
    		값에 따라 분기 처리, break 필수
    while 
    while (조건) {
  // 조건이 true일 동안 반복
}
    		조건이 true인 동안 반복 수행
    do ~ while 
    do {
  // 최소 1회 실행 후 조건 확인
} while (조건);
    		일단 1번 실행하고 조건을 평가
    for 
    for (초기값; 조건; 증감) {
  // 반복 실행
}
    		가장 일반적인 반복문 형태
    for...in 
    for (let key in 객체) {
  // 객체의 속성 이름 반복
}
    		객체의 키(key) 순회
    for...of 
    for (let item of 배열) {
  // 배열 요소 반복
}
    		이터러블 객체(Array, Set 등) 순회
    break / continue 
    for (let i = 0; i < 5; i++) {
  if (i === 3) break;     // 반복 종료
  if (i === 1) continue;  // 이후 문장 건너뜀
}
    		break는 반복 종료,
    continue는 이번 루프 건너뜀
